## DCF 분석

<MultipleChoice
  question="FCFF를 할인할 때 사용하는 할인율은?"
  options={["자기자본비용(Ke)", "타인자본비용(Kd)", "WACC", "무위험이자율(Rf)"]}
  answer={2}
  explanation="FCFF는 채권자와 주주 모두에게 귀속되는 현금흐름이므로 가중평균자본비용(WACC)으로 할인합니다. 반면 FCFE는 주주에게만 귀속되므로 자기자본비용(Ke)으로 할인합니다."
/>

<MultipleChoice
  question="터미널 가치(Terminal Value)가 전체 기업가치에서 차지하는 비중은 일반적으로?"
  options={["10\~20%", "30\~40%", "60\~80%", "90% 이상"]}
  answer={2}
  explanation="터미널 가치는 명시적 추정기간 이후의 가치를 반영하며, 전체 기업가치의 60\~80%를 차지하는 경우가 많습니다. 따라서 터미널 가치 산정에 사용되는 영구성장률과 출구배수의 가정이 매우 중요합니다."
/>

<TrueFalse
  statement="영구성장률은 장기 GDP 성장률보다 높게 설정할 수 있습니다."
  answer={false}
  explanation="영구성장률이 GDP 성장률을 영구적으로 초과하면, 그 기업의 규모가 결국 전체 경제를 초과하게 되므로 논리적으로 불가능합니다. 일반적으로 인플레이션율(2\~3%) 수준에서 설정합니다."
/>

<CalculationExercise
  question="5년차 FCFF가 1,000억 원이고 영구성장률 2%, WACC 10%일 때 터미널 가치는?"
  answer={12750}
  unit="억 원"
  hint="TV = FCFF × (1+g) / (WACC - g)"
  explanation="터미널 가치 = 1,000 × (1 + 0.02) / (0.10 - 0.02) = 1,020 / 0.08 = 12,750억 원. Gordon Growth Model을 적용합니다. 이 값은 5년차 시점의 가치이므로 현재가치로 다시 할인해야 합니다."
/>

<CalculationExercise
  question="자기자본비용 12%, 타인자본비용(세후) 4%, 자기자본 비중 60%, 타인자본 비중 40%일 때 WACC는?"
  answer={8.8}
  unit="%"
  tolerance={0.05}
  hint="WACC = Ke × We + Kd(1-t) × Wd"
  explanation="WACC = 12% × 0.6 + 4% × 0.4 = 7.2% + 1.6% = 8.8%. 자본구조는 시장가치 기준으로 산출해야 합니다."
/>

## 상대가치평가

<QuizContainer title="상대가치평가" totalQuestions={5}>

<MultipleChoice
  question="PER가 높은 기업에 대한 올바른 해석은?"
  options={[
    "항상 고평가된 기업",
    "이익 성장 기대가 높거나 위험이 낮은 기업",
    "자산가치가 높은 기업",
    "배당수익률이 높은 기업"
  ]}
  answer={1}
  explanation="PER = 주가 / EPS. 이론적으로 PER = (1 - b) / (r - g)이므로, 성장률(g)이 높거나 할인율(r)이 낮을수록 PER이 높아집니다. 단순히 높은 PER만으로 고평가라고 판단할 수 없습니다."
/>

<MultipleChoice
  question="EV/EBITDA가 PER보다 유용한 경우는?"
  options={[
    "소규모 벤처기업 비교",
    "자본구조가 다른 기업 간 비교",
    "배당정책 비교",
    "성장률 비교"
  ]}
  answer={1}
  explanation="EV/EBITDA는 자본구조(부채 비율)의 영향을 제거하므로, 레버리지 수준이 다른 기업들을 비교할 때 유용합니다. EV가 기업 전체 가치를, EBITDA가 자본구조와 세금 영향을 제거한 영업수익력을 반영하기 때문입니다."
/>

<TrueFalse
  statement="PBR이 1 미만이면 해당 기업은 반드시 저평가된 것입니다."
  answer={false}
  explanation="PBR < 1은 ROE가 자기자본비용보다 낮을 때 이론적으로 정당화됩니다. 즉, 기업이 자본비용만큼의 수익도 창출하지 못하면 장부가 이하로 거래되는 것이 합리적입니다. Value Trap의 가능성도 고려해야 합니다."
/>

<CalculationExercise
  question="주가 50,000원, EPS 2,500원, BPS 40,000원일 때 PEG(PER/EPS성장률)는? (단, EPS 성장률 25%)"
  answer={0.8}
  tolerance={0.05}
  hint="PER = 주가/EPS, PEG = PER / EPS 성장률(%)"
  explanation="PER = 50,000 / 2,500 = 20배. PEG = 20 / 25 = 0.8. 피터 린치는 PEG < 1이면 저평가, > 1이면 고평가로 판단했습니다."
/>

<CalculationExercise
  question="시가총액 2조 원, 순차입금 5,000억 원, EBITDA 3,000억 원일 때 EV/EBITDA는?"
  answer={8.33}
  unit="배"
  tolerance={0.05}
  hint="EV = 시가총액 + 순차입금"
  explanation="EV = 20,000 + 5,000 = 25,000억 원. EV/EBITDA = 25,000 / 3,000 = 8.33배. 한국 시장 평균은 약 6\~8배이므로 약간 프리미엄이 붙어 있는 수준입니다."
/>

</QuizContainer>
